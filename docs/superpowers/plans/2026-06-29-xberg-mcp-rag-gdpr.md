## Important Notes

### Critical Architecture Decision: New `xberg-rag-node` Crate

The existing `crates/xberg-node/src/lib.rs` is **auto-generated by alef** and cannot be modified directly. Therefore, we create a **new crate** `crates/xberg-rag-node/` for RAG-specific NAPI bindings.

### Existing NAPI Bindings (Already Available)

The following functions are **already exposed** in `xberg-node/lib.rs`:
- `extract()` / `extractBatch()` — document extraction (lines 8951-8978)
- `JsExtractionConfig`, `JsEmbeddingConfig`, `JsRedactionConfig`, `JsNerConfig` — config types
- `extractionConfigDefault()`, `embeddingConfigDefault()`, etc. — default constructors

### xberg-rag Limitations

- `RetrieveMode` enum only has `Vector`, `FullText`, `Hybrid` — **no `Graph` variant yet**
- `RetrieveQuery` has no `graph_depth` field — **needs to be added**
- `VectorStore` trait has no graph operations — **graphqlite integration is separate**

```rust
// crates/xberg-rag/src/backends/graphqlite.rs
use crate::{RagError, RagResult};
use rusqlite::Connection;

pub struct GraphStore {
    conn: graphqlite::Connection,
}

impl GraphStore {
    pub fn new(conn: Connection) -> RagResult<Self> {
        let graph_conn = graphqlite::Connection::from_rusqlite(conn)
            .map_err(|e| RagError::Backend(Box::new(e)))?;
        Ok(Self { conn: graph_conn })
    }
    
    pub fn run_cypher(&self, query: &str) -> RagResult<serde_json::Value> {
        let result = self.conn.run(query)
            .map_err(|e| RagError::Backend(Box::new(e)))?;
        Ok(serde_json::to_value(result).unwrap_or_default())
    }
    
    pub fn louvain(&self, resolution: f64) -> RagResult<Vec<Community>> {
        let communities = self.conn.g.louvain(resolution)
            .map_err(|e| RagError::Backend(Box::new(e)))?;
        Ok(communities)
    }
    
    pub fn pagerank(&self) -> RagResult<Vec<(String, f64)>> {
        let scores = self.conn.g.pagerank()
            .map_err(|e| RagError::Backend(Box::new(e)))?;
        Ok(scores)
    }
    
    pub fn traverse(&self, start_ids: &[String], depth: u32) -> RagResult<Vec<String>> {
        // BFS traversal from seed nodes
        let query = format!(
            "MATCH (d:Document)-[*1..{}]-(related:Document) WHERE d.id IN $start_ids RETURN related.id",
            depth
        );
        let result = self.conn.run(&query)
            .map_err(|e| RagError::Backend(Box::new(e)))?;
        // Parse result to extract related document IDs
        Ok(vec![])
    }
}

#[derive(serde::Serialize, serde::Deserialize)]
pub struct Community {
    pub id: u32,
    pub nodes: Vec<String>,
}
```

```rust
// crates/xberg-rag-node/src/lib.rs
use napi_derive::napi;
use std::sync::Arc;
use xberg_rag::{VectorStore, SqliteVectorStore, CollectionSpec, RetrieveQuery};

#[napi]
pub struct RagStore {
    inner: Arc<dyn VectorStore>,
}

#[napi]
impl RagStore {
    #[napi(factory)]
    pub async fn open_sqlite(name: String, db_path: String) -> napi::Result<Self> {
        let store = SqliteVectorStore::open(name, &db_path).await
            .map_err(|e| napi::Error::from_reason(format!("Failed to open store: {e}")))?;
        Ok(Self { inner: Arc::new(store) })
    }
    
    #[napi]
    pub async fn ensure_collection(&self, spec_json: String) -> napi::Result<()> {
        let spec: CollectionSpec = serde_json::from_str(&spec_json)
            .map_err(|e| napi::Error::from_reason(format!("Invalid spec: {e}")))?;
        self.inner.ensure_collection(&spec).await
            .map_err(|e| napi::Error::from_reason(format!("ensure_collection failed: {e}")))
    }
    
    #[napi]
    pub async fn drop_collection(&self, collection: String) -> napi::Result<()> {
        self.inner.drop_collection(&collection).await
            .map_err(|e| napi::Error::from_reason(format!("drop_collection failed: {e}")))
    }
    
    #[napi]
    pub async fn get_collection(&self, collection: String) -> napi::Result<Option<String>> {
        let spec = self.inner.get_collection(&collection).await
            .map_err(|e| napi::Error::from_reason(format!("get_collection failed: {e}")))?;
        Ok(spec.map(|s| serde_json::to_string(&s).unwrap_or_default()))
    }
    
    #[napi]
    pub async fn upsert_document(
        &self,
        collection: String,
        document_json: String,
        chunks_json: String,
    ) -> napi::Result<String> {
        let doc: xberg_rag::DocumentRecord = serde_json::from_str(&document_json)
            .map_err(|e| napi::Error::from_reason(format!("Invalid document: {e}")))?;
        let chunks: Vec<xberg_rag::ChunkRecord> = serde_json::from_str(&chunks_json)
            .map_err(|e| napi::Error::from_reason(format!("Invalid chunks: {e}")))?;
        
        let doc_id = self.inner.upsert_document(&collection, &doc, &chunks).await
            .map_err(|e| napi::Error::from_reason(format!("upsert_document failed: {e}")))?;
        Ok(doc_id.to_string())
    }
    
    #[napi]
    pub async fn retrieve(
        &self,
        collection: String,
        query_json: String,
    ) -> napi::Result<String> {
        let query: RetrieveQuery = serde_json::from_str(&query_json)
            .map_err(|e| napi::Error::from_reason(format!("Invalid query: {e}")))?;
        
        let output = self.inner.retrieve(&collection, &query).await
            .map_err(|e| napi::Error::from_reason(format!("retrieve failed: {e}")))?;
        
        serde_json::to_string(&output)
            .map_err(|e| napi::Error::from_reason(format!("Serialization failed: {e}")))
    }
    
    #[napi]
    pub async fn delete_documents(&self, collection: String, ids_json: String) -> napi::Result<u64> {
        let ids: Vec<String> = serde_json::from_str(&ids_json)
            .map_err(|e| napi::Error::from_reason(format!("Invalid ids: {e}")))?;
        let doc_ids: Vec<xberg_rag::DocumentId> = ids.into_iter().map(xberg_rag::DocumentId).collect();
        self.inner.delete_documents(&collection, &doc_ids).await
            .map_err(|e| napi::Error::from_reason(format!("delete_documents failed: {e}")))
    }
    
    #[napi]
    pub async fn delete_by_filter(&self, collection: String, filter_json: String) -> napi::Result<u64> {
        let filter: xberg_rag::Filter = serde_json::from_str(&filter_json)
            .map_err(|e| napi::Error::from_reason(format!("Invalid filter: {e}")))?;
        self.inner.delete_by_filter(&collection, &filter).await
            .map_err(|e| napi::Error::from_reason(format!("delete_by_filter failed: {e}")))
    }
    
    #[napi]
    pub async fn collection_stats(&self, collection: String) -> napi::Result<String> {
        let stats = self.inner.collection_stats(&collection).await
            .map_err(|e| napi::Error::from_reason(format!("collection_stats failed: {e}")))?;
        serde_json::to_string(&stats)
            .map_err(|e| napi::Error::from_reason(format!("Serialization failed: {e}")))
    }
}

// Embedding functions (standalone, not through pipeline)
#[napi]
pub async fn embed_texts(texts_json: String, config_json: String) -> napi::Result<String> {
    let texts: Vec<String> = serde_json::from_str(&texts_json)
        .map_err(|e| napi::Error::from_reason(format!("Invalid texts: {e}")))?;
    let config: xberg::EmbeddingConfig = serde_json::from_str(&config_json)
        .map_err(|e| napi::Error::from_reason(format!("Invalid config: {e}")))?;
    
    let embeddings = xberg::embed_texts_async(texts, &config).await
        .map_err(|e| napi::Error::from_reason(format!("Embedding failed: {e}")))?;
    
    serde_json::to_string(&embeddings)
        .map_err(|e| napi::Error::from_reason(format!("Serialization failed: {e}")))
}

// NER function (wraps xberg's detect_entities)
#[napi]
pub async fn detect_entities(text: String, config_json: String) -> napi::Result<String> {
    let config: xberg::NerConfig = serde_json::from_str(&config_json)
        .map_err(|e| napi::Error::from_reason(format!("Invalid config: {e}")))?;
    
    // Create NER backend from config
    let backend = xberg::create_ner_backend(&config).await
        .map_err(|e| napi::Error::from_reason(format!("Failed to create backend: {e}")))?;
    
    let categories = vec![
        xberg::EntityCategory::Person,
        xberg::EntityCategory::Organization,
        xberg::EntityCategory::Location,
    ];
    
    let result = xberg::detect_entities(&text, backend.as_ref(), &categories).await
        .map_err(|e| napi::Error::from_reason(format!("NER failed: {e}")))?;
    
    serde_json::to_string(&result)
        .map_err(|e| napi::Error::from_reason(format!("Serialization failed: {e}")))
}

// Redaction function (wraps xberg's redact)
#[napi]
pub async fn redact_document(document_json: String, config_json: String) -> napi::Result<String> {
    let mut doc: xberg::ExtractedDocument = serde_json::from_str(&document_json)
        .map_err(|e| napi::Error::from_reason(format!("Invalid document: {e}")))?;
    let config: xberg::RedactionConfig = serde_json::from_str(&config_json)
        .map_err(|e| napi::Error::from_reason(format!("Invalid config: {e}")))?;
    
    xberg::redact(&mut doc, &config).await
        .map_err(|e| napi::Error::from_reason(format!("Redaction failed: {e}")))?;
    
    serde_json::to_string(&doc)
        .map_err(|e| napi::Error::from_reason(format!("Serialization failed: {e}")))
}
```

```typescript
// mcp-server/src/tools/ingest.ts
import { z } from "zod";
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import * as fs from "fs";
import * as path from "path";
import { detectPii, PiiFinding } from "../redaction/detect.js";
import { applyTokenReplace, RedactionResult } from "../redaction/redact.js";
import { createRedactedDocx } from "../redaction/output/docx.js";
import { createRedactedPdf } from "../redaction/output/pdf.js";
import { createRedactedText } from "../redaction/output/text.js";
import { createRedactionReport } from "../redaction/output/report.js";
import { RehydrationManager } from "../redaction/rehydration.js";

export function registerIngestTools(server: McpServer) {
  server.tool(
    "ingest_document",
    "Chunk, embed, and store a document in RAG corpus",
    {
      collection: z.string(),
      full_text: z.string(),
      title: z.string().optional(),
      mime: z.string().optional(),
      source_uri: z.string().optional(),
      metadata: z.record(z.any()).optional(),
    },
    async ({ collection, full_text, title, mime, source_uri, metadata }) => {
      return {
        content: [{ type: "text" as const, text: JSON.stringify({ status: "ok" }) }],
      };
    }
  );

  server.tool(
    "ingest_folder",
    "Watch/ingest a folder: extract, redact, create redacted copies + reports, vectorize",
    {
      source_folder: z.string().describe("Path to original files"),
      redacted_folder: z.string().describe("Path for redacted copies"),
      collection: z.string().describe("RAG collection name"),
      mode: z.enum(["one-shot", "watch"]).default("one-shot"),
      redaction_strategy: z.enum(["token_replace", "mask", "hash"]).default("token_replace"),
      preserve_structure: z.boolean().default(true),
    },
    async ({ source_folder, redacted_folder, collection, mode, redaction_strategy, preserve_structure }) => {
      const rehydrationDir = path.join(redacted_folder, ".rehydration");
      const rehydrationManager = new RehydrationManager(rehydrationDir);
      
      // Scan source folder
      const files = fs.readdirSync(source_folder);
      const supportedExts = [".pdf", ".docx", ".txt", ".md", ".html"];
      const supportedFiles = files.filter(f => 
        supportedExts.some(ext => f.toLowerCase().endsWith(ext))
      );
      
      const results: any[] = [];
      
      for (const file of supportedFiles) {
        const filePath = path.join(source_folder, file);
        const ext = path.extname(file).toLowerCase();
        const baseName = path.basename(file, ext);
        
        // 1. Extract text
        const text = fs.readFileSync(filePath, "utf-8");
        
        // 2. Detect PII
        const findings = await detectPii(text);
        
        // 3. Redact
        const redactionResult: RedactionResult = applyTokenReplace(text, findings);
        
        // 4. Create redacted copy
        const redactedFileName = `${baseName}_REDACTED${ext}`;
        const redactedPath = path.join(redacted_folder, redactedFileName);
        
        if (ext === ".docx") {
          await createRedactedDocx(filePath, redactedPath, findings);
        } else if (ext === ".pdf") {
          await createRedactedPdf(filePath, redactedPath, findings);
        } else {
          createRedactedText(filePath, redactedPath, findings);
        }
        
        // 5. Generate report
        const reportFileName = `${baseName}_REPORT.docx`;
        const reportPath = path.join(redacted_folder, reportFileName);
        const reportBuffer = await createRedactionReport({
          documentName: file,
          timestamp: new Date().toISOString(),
          totalRedacted: findings.length,
          findings,
          pages: [{ pageNum: 1, text: redactionResult.redactedText }],
        });
        fs.writeFileSync(reportPath, reportBuffer);
        
        // 6. Store rehydration map
        const docId = `${collection}-${baseName}`;
        await rehydrationManager.saveMap(docId, redactionResult.rehydrationMap, "default-passphrase");
        
        // 7. Ingest into RAG (stub for now)
        // TODO: Call RagStore.upsert_document()
        
        results.push({
          original: filePath,
          redacted: redactedPath,
          report: reportPath,
          rehydration_map: path.join(rehydrationDir, `${docId}.map`),
          pii_count: findings.length,
        });
      }
      
      // Generate summary report
      const summaryPath = path.join(redacted_folder, "SUMMARY_REPORT.docx");
      // TODO: Generate summary report
      
      return {
        content: [{
          type: "text" as const,
          text: JSON.stringify({
            status: "success",
            files_processed: results.length,
            total_redactions: results.reduce((sum, r) => sum + r.pii_count, 0),
            redacted_files: results,
            output_locations: {
              redacted: redacted_folder,
              reports: `${redacted_folder}/*_REPORT.docx`,
              summary: summaryPath,
              rehydration_maps: rehydrationDir,
            },
          }),
        }],
      };
    }
  );
}
```

```typescript
import * as fs from "fs";
import * as path from "path";

export class WarmupManager {
  private cacheDir: string;
  private models: { name: string; path: string; size: number }[];
  
  constructor(cacheDir: string) {
    this.cacheDir = cacheDir;
    this.models = [
      { name: "BGE-M3", path: "embeddings/BAAI--bge-m3/model.onnx", size: 2290000000 },
      { name: "bge-reranker-base", path: "reranker/BAAI--bge-reranker-base/model.onnx", size: 280000000 },
      { name: "GLiNER2-PII", path: "ner/okasi--gliner2-privacy-filter-pii-multi-onnx/model.onnx", size: 510000000 },
    ];
  }
  
  checkModels(): { name: string; cached: boolean; path: string }[] {
    return this.models.map(m => ({
      name: m.name,
      cached: fs.existsSync(path.join(this.cacheDir, m.path)),
      path: m.path,
    }));
  }
  
  getMissingModels(): string[] {
    return this.models
      .filter(m => !fs.existsSync(path.join(this.cacheDir, m.path)))
      .map(m => m.name);
  }
  
  async warmModels(
    progressCallback?: (message: string, progress?: number, total?: number) => void
  ): Promise<void> {
    const missing = this.getMissingModels();
    if (missing.length === 0) {
      progressCallback?.("All models cached");
      return;
    }
    
    for (const modelName of missing) {
      const model = this.models.find(m => m.name === modelName)!;
      progressCallback?.(`Downloading ${model.name}...`, 0, model.size);
      // Trigger download via NAPI-RS
    }
    
    progressCallback?.("All models downloaded and cached");
  }
}
```

```typescript
export async function createRedactionReport(report: RedactionReportData): Promise<Buffer> {

              new TextRun(" of "),
              new TextRun({ children: [PageNumber.TOTAL_PAGES] }),
            ],
          })],
        }),
      },
      children: [
        new Paragraph({
          heading: HeadingLevel.TITLE,
          children: [new TextRun({ text: "PII Detection Report", bold: true, size: 48 })],

function createSummaryTable(findings: PiiFinding[]): Table {
  const categories = [...new Set(findings.map(f => f.category))];
  
  return new Table({
    width: { size: 100, type: WidthType.PERCENTAGE },
    rows: [
      new TableRow({
        children: ["Category", "Count", "Example"].map(h =>
          new TableCell({
            children: [new Paragraph({ children: [new TextRun({ text: h, bold: true })] })],
            shading: { fill: "D9E2F3" },
          })
        ),
      }),
      ...categories.map(cat => {
        const catFindings = findings.filter(f => f.category === cat);
        const color = CATEGORY_COLORS[cat] || { text: "000000", highlight: "white" };
        return new TableRow({
          children: [
            new TableCell({
              children: [new Paragraph({
                children: [new TextRun({ text: cat, color: color.text, bold: true })],
              })],
            }),
            new TableCell({
              children: [new Paragraph({ children: [new TextRun({ text: catFindings.length.toString() })] })],
            }),
            new TableCell({
              children: [new Paragraph({
                children: [new TextRun({ text: catFindings[0]?.token || "N/A", highlight: color.highlight as any })],
              })],
            }),
          ],
        });
      }),
    ],
  });
}

function createHighlightedParagraphs(text: string): Paragraph[] {
  const parts = text.split(/(\[[A-Z]+_\d+\])/);
  
  return [new Paragraph({
    children: parts.map(part => {
      const match = part.match(/^\[([A-Z]+)_(\d+)\]$/);
      if (match) {
        const [, category] = match;
        const color = CATEGORY_COLORS[category] || { text: "000000", highlight: "white" };
        return new TextRun({
          text: part,
          bold: true,
          color: color.text,
          highlight: color.highlight as any,
        });
      }
      return new TextRun({ text: part });
    }),
  })];
}
```

```typescript
interface RedactionFinding {
  token: string;
  original: string;
  category: string;
  startOffset: number;
  endOffset: number;
}

export function createRedactedText(
  originalPath: string,
  redactedPath: string,
  redactions: RedactionFinding[]
): void {
  const text = fs.readFileSync(originalPath, "utf-8");
  let redactedText = text;
  const sortedRedactions = [...redactions].sort((a, b) => b.startOffset - a.startOffset);
  
  for (const finding of sortedRedactions) {
    redactedText = 
      redactedText.slice(0, finding.startOffset) + 
      finding.token + 
      redactedText.slice(finding.endOffset);
  }
  
  fs.writeFileSync(redactedPath, redactedText);
}
```

```typescript
// mcp-server/src/redaction/redact.ts
import { PiiFinding } from "./detect.js";

export interface RedactionResult {
  redactedText: string;
  findings: PiiFinding[];
  rehydrationMap: Map<string, string>;
}

export function applyTokenReplace(
  text: string,
  findings: PiiFinding[]
): RedactionResult {
  const rehydrationMap = new Map<string, string>();
  let redactedText = text;
  
  const sorted = [...findings].sort((a, b) => b.byteOffset - a.byteOffset);
  
  for (const finding of sorted) {
    const token = `[${finding.category}_${rehydrationMap.size + 1}]`;
    rehydrationMap.set(token, finding.original);
    redactedText = redactedText.replace(finding.original, token);
  }
  
  return { redactedText, findings, rehydrationMap };
}
```

```typescript
// mcp-server/src/redaction/rehydration.ts
import * as fs from "fs";
import * as path from "path";

export class RehydrationManager {
  private mapDir: string;
  
  constructor(mapDir: string) {
    this.mapDir = mapDir;
    fs.mkdirSync(mapDir, { recursive: true });
  }
  
  async saveMap(
    documentId: string,
    map: Map<string, string>,
    passphrase: string
  ): Promise<void> {
    const mapPath = path.join(this.mapDir, `${documentId}.map`);
    const data = JSON.stringify(Object.fromEntries(map));
    fs.writeFileSync(mapPath, data); // TODO: encrypt with age/gpg
  }
  
  async loadMap(
    documentId: string,
    passphrase: string
  ): Promise<Map<string, string>> {
    const mapPath = path.join(this.mapDir, `${documentId}.map`);
    if (!fs.existsSync(mapPath)) {
      throw new Error(`Rehydration map not found for ${documentId}`);
    }
    const data = fs.readFileSync(mapPath, "utf-8");
    return new Map(Object.entries(JSON.parse(data)));
  }
  
  async rehydrateToken(
    documentId: string,
    token: string,
    passphrase: string
  ): Promise<string> {
    const map = await this.loadMap(documentId, passphrase);
    const original = map.get(token);
    if (!original) {
      throw new Error(`Token ${token} not found in rehydration map`);
    }
    return original;
  }
}
```
